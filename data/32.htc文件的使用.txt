// supportsVml() borrowed from http://stackoverflow.com/questions/654112/how-do-you-detect-support-for-vml-or-svg-in-a-browser
function supportsVml() {
	if (typeof supportsVml.supported == &quot;undefined&quot;) {
		var a = document.body.appendChild(document.createElement(&apos;div&apos;));
		a.innerHTML = &apos;&lt;v:shape id=&quot;vml_flag1&quot; adj=&quot;1&quot; /&gt;&apos;;
		var b = a.firstChild;
		b.style.behavior = &quot;url(#default#VML)&quot;;
		supportsVml.supported = b ? typeof b.adj == &quot;object&quot;: true;
		a.parentNode.removeChild(a);
	}
	return supportsVml.supported
}


// findPos() borrowed from http://www.quirksmode.org/js/findpos.html
function findPos(obj) {
	var curleft = curtop = 0;

	if (obj.offsetParent) {
		do {
			curleft += obj.offsetLeft;
			curtop += obj.offsetTop;
		} while (obj = obj.offsetParent);
	}

	return({
		&apos;x&apos;: curleft,
		&apos;y&apos;: curtop
	});
}

function createBoxShadow(element, vml_parent) {
	var style = element.currentStyle[&apos;iecss3-box-shadow&apos;] || element.currentStyle[&apos;-moz-box-shadow&apos;] || element.currentStyle[&apos;-webkit-box-shadow&apos;] || element.currentStyle[&apos;box-shadow&apos;] || &apos;&apos;;
	var match = style.match(/^(\d+)px (\d+)px (\d+)px/);
	if (!match) { return(false); }


	var shadow = document.createElement(&apos;v:roundrect&apos;);
	shadow.userAttrs = {
		&apos;x&apos;: parseInt(RegExp.$1 || 0),
		&apos;y&apos;: parseInt(RegExp.$2 || 0),
		&apos;radius&apos;: parseInt(RegExp.$3 || 0) / 2
	};
	shadow.position_offset = {
		&apos;y&apos;: (0 - vml_parent.pos_ieCSS3.y - shadow.userAttrs.radius + shadow.userAttrs.y),
		&apos;x&apos;: (0 - vml_parent.pos_ieCSS3.x - shadow.userAttrs.radius + shadow.userAttrs.x)
	};
	shadow.size_offset = {
		&apos;width&apos;: 0,
		&apos;height&apos;: 0
	};
	shadow.arcsize = element.arcSize +&apos;px&apos;;
	shadow.style.display = &apos;block&apos;;
	shadow.style.position = &apos;absolute&apos;;
	shadow.style.top = (element.pos_ieCSS3.y + shadow.position_offset.y) +&apos;px&apos;;
	shadow.style.left = (element.pos_ieCSS3.x + shadow.position_offset.x) +&apos;px&apos;;
	shadow.style.width = element.offsetWidth +&apos;px&apos;;
	shadow.style.height = element.offsetHeight +&apos;px&apos;;
	shadow.style.antialias = true;
	shadow.className = &apos;vml_box_shadow&apos;;
	shadow.style.zIndex = element.zIndex - 1;
	shadow.style.filter = &apos;progid:DXImageTransform.Microsoft.Blur(pixelRadius=&apos;+ shadow.userAttrs.radius +&apos;,makeShadow=true,shadowOpacity=&apos;+ element.opacity +&apos;)&apos;;

	element.parentNode.appendChild(shadow);
	//element.parentNode.insertBefore(shadow, element.element);

	// For window resizing
	element.vml.push(shadow);

	return(true);
}

function createBorderRect(element, vml_parent) {
	if (isNaN(element.borderRadius)) { return(false); }

	element.style.background = &apos;transparent&apos;;
	element.style.borderColor = &apos;transparent&apos;;

	var rect = document.createElement(&apos;v:roundrect&apos;);
	rect.position_offset = {
		&apos;y&apos;: (0.5 * element.strokeWeight) - vml_parent.pos_ieCSS3.y,
		&apos;x&apos;: (0.5 * element.strokeWeight) - vml_parent.pos_ieCSS3.x
	};
	rect.size_offset = {
		&apos;width&apos;: 0 - element.strokeWeight,
		&apos;height&apos;: 0 - element.strokeWeight
	};
	rect.arcsize = element.arcSize +&apos;px&apos;;
	rect.strokeColor = element.strokeColor;
	rect.strokeWeight = element.strokeWeight +&apos;px&apos;;
	rect.stroked = element.stroked;
	rect.className = &apos;vml_border_radius&apos;;
	rect.style.display = &apos;block&apos;;
	rect.style.position = &apos;absolute&apos;;
	rect.style.top = (element.pos_ieCSS3.y + rect.position_offset.y) +&apos;px&apos;;
	rect.style.left = (element.pos_ieCSS3.x + rect.position_offset.x) +&apos;px&apos;;
	rect.style.width = (element.offsetWidth + rect.size_offset.width) +&apos;px&apos;;
	rect.style.height = (element.offsetHeight + rect.size_offset.height) +&apos;px&apos;;
	rect.style.antialias = true;
	rect.style.zIndex = element.zIndex - 1;

	if (border_opacity &amp;&amp; (element.opacity &lt; 1)) {
		rect.style.filter = &apos;progid:DXImageTransform.Microsoft.Alpha(Opacity=&apos;+ parseFloat(element.opacity * 100) +&apos;)&apos;;
	}

	var fill = document.createElement(&apos;v:fill&apos;);
	fill.color = element.fillColor;
	fill.src = element.fillSrc;
	fill.className = &apos;vml_border_radius_fill&apos;;
	fill.type = &apos;tile&apos;;
	fill.opacity = element.opacity;

	// Hack: IE6 doesn&apos;t support transparent borders, use padding to offset original element
	isIE6 = /msie|MSIE 6/.test(navigator.userAgent);
	if (isIE6 &amp;&amp; (element.strokeWeight &gt; 0)) {
		element.style.borderStyle = &apos;none&apos;;
		element.style.paddingTop = parseInt(element.currentStyle.paddingTop || 0) + element.strokeWeight;
		element.style.paddingBottom = parseInt(element.currentStyle.paddingBottom || 0) + element.strokeWeight;
	}

	rect.appendChild(fill);
	element.parentNode.appendChild(rect);
	//element.parentNode.insertBefore(rect, element.element);

	// For window resizing
	element.vml.push(rect);

	return(true);
}

function createTextShadow(element, vml_parent) {
	if (!element.textShadow) { return(false); }

	var match = element.textShadow.match(/^(\d+)px (\d+)px (\d+)px (#?\w+)/);
	if (!match) { return(false); }


	//var shadow = document.createElement(&apos;span&apos;);
	var shadow = element.cloneNode(true);
	var radius = parseInt(RegExp.$3 || 0);
	shadow.userAttrs = {
		&apos;x&apos;: parseInt(RegExp.$1 || 0) - (radius),
		&apos;y&apos;: parseInt(RegExp.$2 || 0) - (radius),
		&apos;radius&apos;: radius / 2,
		&apos;color&apos;: (RegExp.$4 || &apos;#000&apos;)
	};
	shadow.position_offset = {
		&apos;y&apos;: (0 - vml_parent.pos_ieCSS3.y + shadow.userAttrs.y),
		&apos;x&apos;: (0 - vml_parent.pos_ieCSS3.x + shadow.userAttrs.x)
	};
	shadow.size_offset = {
		&apos;width&apos;: 0,
		&apos;height&apos;: 0
	};
	shadow.style.color = shadow.userAttrs.color;
	shadow.style.position = &apos;absolute&apos;;
	shadow.style.top = (element.pos_ieCSS3.y + shadow.position_offset.y) +&apos;px&apos;;
	shadow.style.left = (element.pos_ieCSS3.x + shadow.position_offset.x) +&apos;px&apos;;
	shadow.style.antialias = true;
	shadow.style.behavior = null;
	shadow.className = &apos;ieCSS3_text_shadow&apos;;
	shadow.innerHTML = element.innerHTML;
	// For some reason it only looks right with opacity at 75%
	shadow.style.filter = &apos;\
		progid:DXImageTransform.Microsoft.Alpha(Opacity=75)\
		progid:DXImageTransform.Microsoft.Blur(pixelRadius=&apos;+ shadow.userAttrs.radius +&apos;,makeShadow=false,shadowOpacity=100)\
	&apos;;

	var clone = element.cloneNode(true);
	clone.position_offset = {
		&apos;y&apos;: (0 - vml_parent.pos_ieCSS3.y),
		&apos;x&apos;: (0 - vml_parent.pos_ieCSS3.x)
	};
	clone.size_offset = {
		&apos;width&apos;: 0,
		&apos;height&apos;: 0
	};
	clone.style.behavior = null;
	clone.style.position = &apos;absolute&apos;;
	clone.style.top = (element.pos_ieCSS3.y + clone.position_offset.y) +&apos;px&apos;;
	clone.style.left = (element.pos_ieCSS3.x + clone.position_offset.x) +&apos;px&apos;;
	clone.className = &apos;ieCSS3_text_shadow&apos;;


	element.parentNode.appendChild(shadow);
	element.parentNode.appendChild(clone);

	element.style.visibility = &apos;hidden&apos;;

	// For window resizing
	element.vml.push(clone);
	element.vml.push(shadow);

	return(true);
}

function ondocumentready(classID) {
	if (!supportsVml()) { return(false); }

  if (this.className.match(classID)) { return(false); }
	this.className = this.className.concat(&apos; &apos;, classID);

	// Add a namespace for VML (IE8 requires it)
	if (!document.namespaces.v) { document.namespaces.add(&quot;v&quot;, &quot;urn:schemas-microsoft-com:vml&quot;); }

	// Check to see if we&apos;ve run once before on this page
	if (typeof(window.ieCSS3) == &apos;undefined&apos;) {
		// Create global ieCSS3 object
		window.ieCSS3 = {
			&apos;vmlified_elements&apos;: new Array(),
			&apos;update_timer&apos;: setInterval(updatePositionAndSize, timer_length)
		};

		if (typeof(window.onresize) == &apos;function&apos;) { window.ieCSS3.previous_onresize = window.onresize; }

		// Attach window resize event
		window.onresize = updatePositionAndSize;
	}


	// These attrs are for the script and have no meaning to the browser:
	this.borderRadius = parseInt(this.currentStyle[&apos;iecss3-border-radius&apos;] ||
	                             this.currentStyle[&apos;-moz-border-radius&apos;] ||
	                             this.currentStyle[&apos;-webkit-border-radius&apos;] ||
	                             this.currentStyle[&apos;border-radius&apos;] ||
	                             this.currentStyle[&apos;-khtml-border-radius&apos;]);
	this.arcSize = Math.min(this.borderRadius / Math.min(this.offsetWidth, this.offsetHeight), 1);
	this.fillColor = this.currentStyle.backgroundColor;
	this.fillSrc = this.currentStyle.backgroundImage.replace(/^url\(&quot;(.+)&quot;\)$/, &apos;$1&apos;);
	this.strokeColor = this.currentStyle.borderColor;
	this.strokeWeight = parseInt(this.currentStyle.borderWidth);
	this.stroked = &apos;true&apos;;
	if (isNaN(this.strokeWeight) || (this.strokeWeight == 0)) {
		this.strokeWeight = 0;
		this.strokeColor = fillColor;
		this.stroked = &apos;false&apos;;
	}
	this.opacity = parseFloat(this.currentStyle.opacity || 1);
	this.textShadow = this.currentStyle[&apos;text-shadow&apos;];

	this.element.vml = new Array();
	this.zIndex = parseInt(this.currentStyle.zIndex);
	if (isNaN(this.zIndex)) { this.zIndex = 0; }

	// Find which element provides position:relative for the target element (default to BODY)
	vml_parent = this;
	var limit = 100, i = 0;
	do {
		vml_parent = vml_parent.parentElement;
		i++;
		if (i &gt;= limit) { return(false); }
	} while ((typeof(vml_parent) != &apos;undefined&apos;) &amp;&amp; (vml_parent.currentStyle.position != &apos;relative&apos;) &amp;&amp; (vml_parent.tagName != &apos;BODY&apos;));

	vml_parent.pos_ieCSS3 = findPos(vml_parent);
	this.pos_ieCSS3 = findPos(this);

	var rv1 = createBoxShadow(this, vml_parent);
	var rv2 = createBorderRect(this, vml_parent);
	var rv3 = createTextShadow(this, vml_parent);
	if (rv1 || rv2 || rv3) { window.ieCSS3.vmlified_elements.push(this.element); }

	if (typeof(vml_parent.document.ieCSS3_stylesheet) == &apos;undefined&apos;) {
		vml_parent.document.ieCSS3_stylesheet = vml_parent.document.createStyleSheet();
		vml_parent.document.ieCSS3_stylesheet.addRule(&quot;v\\:roundrect&quot;, &quot;behavior: url(#default#VML)&quot;);
		vml_parent.document.ieCSS3_stylesheet.addRule(&quot;v\\:fill&quot;, &quot;behavior: url(#default#VML)&quot;);
		// Compatibility with IE7.js
		vml_parent.document.ieCSS3_stylesheet.ie7 = true;
	}
}

function updatePositionAndSize() {
	if (typeof(window.ieCSS3.vmlified_elements) != &apos;object&apos;) { return(false); }

	for (var i in window.ieCSS3.vmlified_elements) {
		var el = window.ieCSS3.vmlified_elements[i];

		if (typeof(el.vml) != &apos;object&apos;) { continue; }

		for (var z in el.vml) {
